# See also http://b/issue?id=7132535.

# Enable IME
SEND_KEY	ON
RESET_CONTEXT

# Set to Qwerty kana mode.

SET_CONFIG	use_kana_modifier_insensitive_conversion	true

# Type "てつとう"
SEND_KEYS	4444
SEND_KEY	RIGHT
SEND_KEYS	444
SEND_KEY	RIGHT
SEND_KEYS	44444111

# The result should be "鉄道" for now.
# TODO(taku): change the default param so literal-on-top stops working.
EXPECT_CANDIDATE	1	鉄道

# Move the caret to the beginnig, and back to the end.
SEND_KEY	LEFT
SEND_KEY	LEFT
SEND_KEY	LEFT
SEND_KEY	RIGHT
SEND_KEY	RIGHT
SEND_KEY	RIGHT

# The candidate is still "鉄道"
# TODO(taku): change the default param so literal-on-top stops working.
EXPECT_CANDIDATE	1	鉄道
